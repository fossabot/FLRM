var crypto = require('crypto')
/**
*@package
*@returns first key
*/
function new_1() {
  var $1 = crypto.createHash('sha256');  
  $1.update('0.4','utf8');
  return $1.digest('base64')
}
/**
*@package
*@returns second key
*/
function new_2() {
  var $2 = crypto.createHash('sha256');
  $2.update('0.4','utf8');
  return $2.digest('hex')
}
/**
*@public
*@param {array} input stated like this:  [KEY1,KEY2]
*@returns {number} 11,0,10,1
*/
module.exports.keyMatch = function(input){
  var arr = [false,false];
  if (input[0] === new_1())
    {
      arr[0] = true;
    }
  else if(input[1] === new_2())
    {
      arr[1] = true;
    }
  if((arr[0] && arr[1]) === true){
    return 11
  }
  if((arr[0] && arr[1]) !== true){
    return parseInt('00')
  }
  if(arr[0] === true && arr[1] === false){
    return 10
  }
  if(arr[1] === true && arr[0] === false){
    return parseInt('01')
  }
}
